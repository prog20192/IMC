peso = float (input ("Digite seu peso:   "))
altura = float (input ("Digita sua altura:  "))
imc = peso/(altura**2)
print ("Seu imc é", imc)

if imc >= 30:
    print ("Muito Acima do Peso")
elif imc >= 25:
    print ("Acima do Peso normal")
elif imc >= 18:
    print ("Normal")
elif imc >= 17:
    print ("Abaixo do peso")
